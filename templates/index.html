import re
import pandas as pd
import matplotlib.pyplot as plt

# =========================
# SETTINGS
# =========================
CHAT_FILE = "WhatsApp Chat with.txt"
MIN_AVG_MESSAGES = 2
TOP_PER_DAY = 5
PIE_CHART_FILE = "pie_chart.png"
STACKED_BAR_FILE = "stacked_bar_chart.png"

# =========================
# REGEX (date + sender)
# =========================
pattern = re.compile(r"(\d{1,2}/\d{1,2}/\d{2}),.*?-\s([^:]+):")

records = []

# =========================
# READ FILE
# =========================
with open(CHAT_FILE, "r", encoding="utf-8") as f:
    for line in f:
        match = pattern.match(line)
        if match:
            date, sender = match.groups()
            records.append({
                "date": date,
                "sender": sender.strip()
            })

# =========================
# CREATE DATAFRAME
# =========================
df = pd.DataFrame(records)
df["date"] = pd.to_datetime(df["date"], format="%d/%m/%y")

# =========================
# DAILY MESSAGE COUNTS
# =========================
counts = df.groupby(["date", "sender"]).size().unstack(fill_value=0)

# =========================
# STATISTICS
# =========================
totals = counts.sum()
avg_per_day = counts.mean()

# Filter senders for pie chart
pie_people = avg_per_day[avg_per_day >= MIN_AVG_MESSAGES].index
pie_totals = totals[pie_people]

# =========================
# STACKED BAR CHART (TOP N per day)
# =========================
NUM_PARTS = 5   # split chart into 3 parts

stack_data = counts.copy()

# Prepare day-month index
stack_data_indexed = stack_data.copy()
stack_data_indexed.index = stack_data_indexed.index.strftime('%d-%m')

# Split into parts
split = len(stack_data_indexed) // NUM_PARTS + 1
parts = [stack_data_indexed.iloc[i*split:(i+1)*split, :] for i in range(NUM_PARTS)]

for i, part in enumerate(parts):
    if part.empty:
        continue

    # Prepare top-N + Others per day
    part_top = pd.DataFrame(index=part.index)
    for day in part.index:
        row = part.loc[day]
        top_senders = row.nlargest(TOP_PER_DAY)
        others_sum = row.drop(top_senders.index).sum()
        for sender, val in top_senders.items():
            part_top.at[day, sender] = val
        if others_sum > 0:
            part_top.at[day, 'Others'] = others_sum
    part_top = part_top.fillna(0)

    # Calculate max total messages for y-axis
    max_total = part_top.sum(axis=1).max()

    # Plot
    fig, ax = plt.subplots(figsize=(12,6))
    part_top.plot(kind='bar', stacked=True, ax=ax, colormap='tab20')

    ax.set_xlabel("Date (day-month)", labelpad=10)
    ax.set_ylabel("Messages", labelpad=10)
    ax.set_title(f"Top {TOP_PER_DAY} + Others Per Day (Part {i+1} of {NUM_PARTS})", pad=15)
    ax.set_ylim(0, max_total * 1.10)  # small padding above the highest stack

    # Annotate total messages on top of each bar
    for idx, day in enumerate(part_top.index):
        total = part_top.loc[day].sum()
        ax.text(idx, total + max_total*0.01, f"{int(total)}", ha='center', va='bottom', fontsize=9, fontweight='bold')

    # Clean legend
    handles, labels = ax.get_legend_handles_labels()
    by_label = dict(zip(labels, handles))
    ax.legend(by_label.values(), by_label.keys(), title="Sender", bbox_to_anchor=(1.05,1), loc='upper left')

    plt.xticks(rotation=90, ha='center', fontsize=9)
    plt.tight_layout()
    fig.savefig(f"stacked_bar_part{i+1}.png", dpi=300, bbox_inches='tight')

import pandas as pd
import matplotlib.pyplot as plt

# Load CSV
df = pd.read_csv("Others/filtered_messages.csv")

# Clean column names
df.columns = df.columns.str.strip().str.lower()

# Convert date
df["date"] = pd.to_datetime(df["date"], format="%d/%m/%y")

# Remove date column
data = df.drop(columns=["date"])

# Total messages per person
totals = data.sum()

person = "random_name"

# Only label random_name
labels = []
for name in totals.index:
    if name == person:
        labels.append("random_name")
    else:
        labels.append("")

fig, ax = plt.subplots()

wedges, texts, autotexts = ax.pie(
    totals,
    labels=labels,
    autopct="%1.1f%%",
    startangle=90
)

# Rotate percentage labels
for autotext in autotexts:
    autotext.set_rotation(-60)

# Donut hole
centre_circle = plt.Circle((0,0),0.70,fc='white')
fig.gca().add_artist(centre_circle)

ax.set_title("Message Share in Chat")

plt.tight_layout()
plt.show()
